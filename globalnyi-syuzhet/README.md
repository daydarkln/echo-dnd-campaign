graph TD

%% Начало воздействия на группу

A --> C[Контакт с Карающей Тенью]:::critical

%% Влияние спор на первого героя
B --> D1[Лёгкое заражение: головокружение, кашель]:::neutral

%% Влияние Тени на второго героя

F1 --> F2[Искушение выполнение морально сомнительных действий]:::hazard

%% Возможные способы избавления от Тени

F2 --> G2[Магическая защита Aura of Purity Mind Blank]:::neutral
F3 --> G3[Разрушение ядра Тени уничтожение её магической сущности]:::boss

%% Взаимодействие заражения и влияния
D3 --> H[Усиление телепатического воздействия на второго героя]:::critical
F3 --> I[Грибная колония активирует дополнительное заражение]:::critical

%% Финальная развязка

G3 --> J
E4 --> J
G4 --> J
